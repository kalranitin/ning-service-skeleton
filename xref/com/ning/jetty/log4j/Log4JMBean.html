17  package com.ning.jetty.log4j;
18  
19  import org.apache.commons.lang.StringUtils;
20  import org.apache.log4j.Level;
21  import org.apache.log4j.LogManager;
22  import org.apache.log4j.Logger;
23  import org.weakref.jmx.Managed;
24  
25  import java.util.ArrayList;
26  import java.util.Collections;
27  import java.util.Date;
28  import java.util.Enumeration;
29  import java.util.List;
30  import java.util.concurrent.atomic.AtomicLong;
31  
32  import static com.ning.jetty.log4j.LogLevelCounter.LevelIndex;
33  
34  public class Log4JMBean
35  {
36      private final AtomicLong lastResetTimeMillis = new AtomicLong(System.currentTimeMillis());
37  
38      private LogLevelCounter counter = null;
39  
40      public Log4JMBean()
41      {
42          initLogLevelCounting();
43      }
44  
45      private void initLogLevelCounting()
46      {
47          this.counter = new LogLevelCounter();
48          LogLevelCounterAppender.registerLogLevelCounter(counter);
49      }
50  
51      @Managed(description = "Set the log level on a per logger basis")
52      public void setLevel(final String loggerName, final String levelString)
53      {
54          final Logger logger;
55  
56          if (!StringUtils.isEmpty(loggerName)) {
57              logger = LogManager.getLogger(loggerName);
58          }
59          else {
60              logger = LogManager.getRootLogger();
61          }
62  
63          logger.setLevel(Level.toLevel(levelString));
64      }
65  
66      @Managed(description = "Get the log level for the specified logger")
67      public String getLevel(final String loggerName)
68      {
69          final Logger logger;
70  
71          if (!StringUtils.isEmpty(loggerName)) {
72              logger = LogManager.getLogger(loggerName);
73          }
74          else {
75              logger = LogManager.getRootLogger();
76          }
77  
78          return logger.getLevel().toString();
79      }
80  
81      @Managed(description = "Retrieve the parent logger name")
82      public String getParentLoggerName(final String loggerName)
83      {
84          if (!StringUtils.isEmpty(loggerName)) {
85              return LogManager.getLogger(loggerName).getParent().getName();
86          }
87  
88          return "";
89      }
90  
91      @Managed(description = "Get the current loggers names")
92      public String[] getLoggerNames()
93      {
94          final Enumeration<Logger> e = LogManager.getCurrentLoggers();
95  
96          final List<String> loggers = new ArrayList<String>();
97  
98          while (e.hasMoreElements()) {
99              final Logger logger = e.nextElement();
100             loggers.add(logger.getName());
101         }
102 
103         Collections.sort(loggers);
104 
105         return loggers.toArray(new String[0]);
106     }
107 
108     @Managed(description = "Get the count of log events for the specified level")
109     public long getLogLevelCountByLevel(final String levelString)
110     {
111         final long[] counts = counter.getLogLevelCounts();
112         final LevelIndex lIndex = LevelIndex.getLevelIndexFromLevelString(levelString);
113         if (lIndex == null) {
114             return -1;
115         }
116         else {
117             return counts[lIndex.getIndex()];
118         }
119     }
120 
121     @Managed(description = "Get the count of log events per level")
122     public String[] getLogLevelCounts()
123     {
124         final long[] counts = counter.getLogLevelCounts();
125 
126         final String[] levelMessages = new String[LevelIndex.getNumLevels()];
127         for (final LevelIndex lIndex : LevelIndex.values()) {
128             final String levelString = lIndex.getLevelString();
129             levelMessages[lIndex.getIndex()] = levelString + ": " + counts[lIndex.getIndex()];
130         }
131 
132         return levelMessages;
133     }
134 
135     @Managed(description = "Reset all counters")
136     public void resetStats()
137     {
138         counter.resetAllLogLevelCounts();
139         lastResetTimeMillis.set(System.currentTimeMillis());
140     }
141 
142     @Managed(description = "Retrieve the last time the counters were reset")
143     public Date getLastStatsResetTime()
144     {
145         return new Date(lastResetTimeMillis.get());
146     }
147 
148     @Managed(description = "Retrieve the status of log counting per level")
149     public String[] getLogLevelCountingEnabled()
150     {
151         final String[] levelMessages = new String[LevelIndex.getNumLevels()];
152         for (final LevelIndex lIndex : LevelIndex.values()) {
153             final String levelString = lIndex.getLevelString();
154             levelMessages[lIndex.getIndex()] = levelString + ": " + counter.getCountingEnabledByLevel(lIndex.getLevel());
155         }
156 
157         return levelMessages;
158     }
159 
160     @Managed(description = "Enable or disable log counting per level")
161     public void setLogLevelCountingEnabledByLevel(final String levelString, final boolean enabled)
162     {
163         final Level level = LevelIndex.getLevelFromLevelString(levelString);
164         counter.setCountingEnabledByLevel(level, enabled);
165     }

181     }

183     @Managed(description = "Get the count of log events for WARN")
184     public long getLogLevelCountForWarn()
185     {

187     }

189     @Managed(description = "Get the count of log events for ERROR")
190     public long getLogLevelCountForError()

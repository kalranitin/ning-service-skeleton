17  package com.ning.jetty.utils.providers;
18  
19  import com.google.inject.Inject;
20  import com.google.inject.Provider;
21  import com.jolbox.bonecp.BoneCPConfig;
22  import com.jolbox.bonecp.BoneCPDataSource;
23  import com.ning.jdbi.metrics.JdbiGroupStrategy;
24  import com.ning.jdbi.metrics.MetricsTimingCollector;
25  import com.ning.jdbi.metrics.SqlJdbiGroupStrategy;
26  import com.ning.jetty.utils.DaoConfig;
27  import com.yammer.metrics.core.MetricsRegistry;
28  import org.skife.jdbi.v2.DBI;
29  import org.skife.jdbi.v2.TimingCollector;
30  import org.skife.jdbi.v2.logging.Log4JLog;
31  import org.skife.jdbi.v2.tweak.SQLLog;
32  
33  import java.util.concurrent.TimeUnit;
34  
35  public class DBIProvider implements Provider<DBI>
36  {
37      private final MetricsRegistry metricsRegistry;
38      private final DaoConfig config;
39  
40      @Inject
41      public DBIProvider(final MetricsRegistry metricsRegistry, final DaoConfig config)
42      {
43          this.metricsRegistry = metricsRegistry;
44          this.config = config;
45      }
46  
47      @Override
48      public DBI get()
49      {
50          final BoneCPConfig dbConfig = new BoneCPConfig();
51          dbConfig.setJdbcUrl(config.getJdbcUrl());
52          dbConfig.setUsername(config.getUsername());
53          dbConfig.setPassword(config.getPassword());
54          dbConfig.setMinConnectionsPerPartition(config.getMinIdle());
55          dbConfig.setMaxConnectionsPerPartition(config.getMaxActive());
56          dbConfig.setConnectionTimeout(config.getConnectionTimeout().getPeriod(), config.getConnectionTimeout().getUnit());
57          dbConfig.setPartitionCount(1);
58          dbConfig.setDefaultTransactionIsolation("READ_COMMITTED");
59          dbConfig.setDisableJMX(false);
60  
61          final BoneCPDataSource ds = new BoneCPDataSource(dbConfig);
62          final DBI dbi = new DBI(ds);
63          final SQLLog log = new Log4JLog();
64          dbi.setSQLLog(log);
65  
66          final JdbiGroupStrategy jdbiGroupStrategy = new SqlJdbiGroupStrategy();
67          final TimingCollector timingCollector = new MetricsTimingCollector(metricsRegistry, jdbiGroupStrategy, TimeUnit.MILLISECONDS, TimeUnit.SECONDS);
68          dbi.setTimingCollector(timingCollector);
69  
70          return dbi;
71      }
72  }

17  package com.ning.jetty.utils.filters;
18  
19  import com.google.inject.Inject;
20  import com.ning.metrics.eventtracker.CollectorController;
21  import com.ning.metrics.eventtracker.smile.org.codehaus.jackson.JsonNode;
22  import com.ning.metrics.eventtracker.smile.org.codehaus.jackson.map.ObjectMapper;
23  import com.ning.metrics.serialization.event.SmileEnvelopeEvent;
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  import java.io.IOException;
28  
29  /**
30   * Tracker implementation that uses the eventtracker library, to send events to the Ning collector. To use it:
31   * <p/>
32   * install(new CollectorControllerSmileModule());
33   * install(new CollectorControllerHttpMBeanModule());
34   * bind(Tracker.class).to(CollectorTracker.class).asEagerSingleton();
35   */
36  public class CollectorTracker implements Tracker
37  {
38      // Make sure to use the eventtracker mapper
39      private static final ObjectMapper mapper = new ObjectMapper();
40  
41      private final Logger log = LoggerFactory.getLogger(CollectorTracker.class);
42      private final CollectorController controller;
43      private final String eventName = System.getProperty("com.ning.core.eventtracker.requestLogEventName", "RequestLogEvent");
44  
45      @Inject
46      public CollectorTracker(final CollectorController controller)
47      {
48          this.controller = controller;
49      }
50  
51      @Override
52      public void trackRequest(final RequestLog request)
53      {
54          final JsonNode root = mapper.valueToTree(request);
55          try {
56              controller.offerEvent(new SmileEnvelopeEvent(eventName, root));
57          }
58          catch (IOException e) {
59              log.warn("Got I/O exception trying to send RequestLog [{}]: {}", request, e.toString());
60          }
61      }
62  }

57  public class BaseServerModule extends ServerModule
58  {
59      private static final List<String> FILTERS = ImmutableList.of(
60          GZIPContentEncodingFilter.class.getName(),
61          LoggingFilter.class.getName()
62      );
63  
64      private static final ImmutableMap.Builder<String, String> JERSEY_PARAMS = new ImmutableMap.Builder<String, String>()
65          .put(PROPERTY_CONTAINER_REQUEST_FILTERS, StringUtils.join(FILTERS, ";"))
66              // Though it would seem to make sense that filters should be applied to responses in reverse order, in fact the
67              // response filters appear to wrap each other up before executing, with the result being that execution order
68              // is the reverse of the declared order.
69          .put(PROPERTY_CONTAINER_RESPONSE_FILTERS, StringUtils.join(FILTERS, ";"));

71      // System properties
72      private final Properties props;
73      // config-magic classes
74      private final ArrayList<Class> configs = new ArrayList<Class>();
75      // Healthcheck classes
76      private final ArrayList<Class<? extends HealthCheck>> healthchecks = new ArrayList<Class<? extends HealthCheck>>();
77      // JMX beans to export
78      private final ArrayList<Class> beans = new ArrayList<Class>();
79      // Arecibo integration
80      private String areciboProfile = null;
81      // eventtracker integration
82      private boolean trackRequests = false;
83      // Whether log4j is used
84      private boolean log4jEnabled = false;
85      // Jersey resources
86      final List<String> resources = new ArrayList<String>();
87      // Extra Guice modules to install
88      final List<Module> modules = new ArrayList<Module>();
89      private Map<String, ArrayList<Map.Entry<Class<? extends Filter>, Map<String, String>>>> filters;
90      private Map<String, Class<? extends HttpServlet>> serves;

92      public BaseServerModule(final List<Class> configs,
93                              final List<Class<? extends HealthCheck>> healthchecks,
94                              final List<Class> beans,
95                              final String areciboProfile,
96                              final boolean trackRequests,
97                              final boolean log4jEnabled,
98                              final List<String> resources,
99                              final List<Module> modules,
100                             final Map<String, ArrayList<Map.Entry<Class<? extends Filter>, Map<String, String>>>> filters,
101                             final Map<String, Class<? extends HttpServlet>> serves)
102     {
103         this.props = System.getProperties();
104         this.configs.addAll(configs);
105         this.healthchecks.addAll(healthchecks);
106         this.beans.addAll(beans);
107         this.areciboProfile = areciboProfile;
108         this.log4jEnabled = log4jEnabled;
109         this.trackRequests = trackRequests;
110         this.resources.addAll(resources);
111         this.modules.addAll(modules);
112         this.filters = filters;
113         this.serves = serves;
114 
115         this.configs.add(DaoConfig.class);
116         this.configs.add(TrackerConfig.class);
117     }

119     @Override
120     public void configureServlets()
121     {
122         super.configureServlets();
123 
124         installHealthChecks();
125         installArecibo();
126         installEventtracker();
127         installLog4j();
128         installExtraModules();
129 
130         configureJersey();
131     }

133     private void installEventtracker()
134     {
135         if (!trackRequests) {
136             return;
137         }
138 
139         install(new CollectorControllerSmileModule());
140         install(new CollectorControllerHttpMBeanModule());
141         bind(Tracker.class).to(CollectorTracker.class).asEagerSingleton();
142         filter("*").through(TrackerFilter.class);
143     }
144 
145     private void installLog4j()
146     {
147         if (!log4jEnabled) {
148             return;
149         }
150 
151         bind(Log4JMBean.class).asEagerSingleton();
152     }
153 
154     @Override
155     protected void configureConfig()
156     {
157         super.configureConfig();
158 
159         for (final Class configClass : configs) {
160             bind(configClass).toInstance(new ConfigurationObjectFactory(props).build(configClass));
161         }
162     }
163 
164     @Override
165     protected void installJMX()
166     {
167         super.installJMX();
168 
169         final ExportBuilder builder = MBeanModule.newExporter(binder());
170         for (final Class beanClass : beans) {
171             builder.export(beanClass).withGeneratedName();
172         }
173 
174         if (log4jEnabled) {
175             builder.export(Log4JMBean.class).withGeneratedName();
176         }
177     }
178 
179     protected void installHealthChecks()
180     {
181         final Multibinder<HealthCheck> healthChecksBinder = Multibinder.newSetBinder(binder(), HealthCheck.class);
182 
183         for (final Class<? extends HealthCheck> healthCheckClass : healthchecks) {
184             healthChecksBinder.addBinding().to(healthCheckClass).asEagerSingleton();
185         }
186     }
187 
188     protected void installArecibo()
189     {
190         if (areciboProfile == null) {
191             return;
192         }
193 
194         install(new AreciboMonitoringModule(areciboProfile));
195         // Expose metrics objects to Arecibo
196         install(new AreciboMetricsModule());
197         bind(Jetty7AreciboConnector.class).asEagerSingleton();
198 
199         if (log4jEnabled) {
200             bind(Log4JMBeanAreciboConnector.class).asEagerSingleton();
201         }
202     }
203 
204     private void installExtraModules()
205     {
206         for (final Module module : modules) {
207             install(module);
208         }
209     }
210 
211     protected void configureJersey()
212     {
213         for (final String urlPattern : filters.keySet()) {
214             for (final Map.Entry<Class<? extends Filter>, Map<String, String>> filter : filters.get(urlPattern)) {
215                 filter(urlPattern).through(filter.getKey(), filter.getValue());
216             }
217         }
218 
219         for (final String urlPattern : serves.keySet()) {
220             serve(urlPattern).with(serves.get(urlPattern), JERSEY_PARAMS.build());
221         }
222 
223         // Catch-all resources
224         if (resources.size() != 0) {
225             JERSEY_PARAMS.put(PROPERTY_PACKAGES, StringUtils.join(resources, ";"));
226             filter("*").through(GuiceContainer.class, JERSEY_PARAMS.build());
227         }
228     }
